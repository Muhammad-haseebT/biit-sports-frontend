5. NO BALL MODAL (NB button)
User selects batsman runs: 0, 1, 2, 3, 4, 6
User checks/unchecks: "Wicket on this ball" (for catches on no-ball)
Display shows: Total = 1 (no-ball) + runs

Example: No Ball + 4 runs by batsman
javascript
{
    matchId: 1,
    inningsId: 1,
    teamId: 1,
    batsmanId: 101,
    nonStrikerId: 102,
    bowlerId: 201,
    
    event: "5",              // Total: 1 no-ball + 4 runs
    eventType: "noball",
    extraType: "no-ball",
    isLegal: false,          // ❌ No-ball is illegal
    runsOnThisBall: 4,       // Batsman scored 4
    extrasThisBall: 1,       // 1 no-ball penalty
    four: true,              // If 4 runs, count as boundary
    six: false
}
Example: No Ball + Wicket (Run Out on no-ball)
javascript
{
    matchId: 1,
    inningsId: 1,
    teamId: 1,
    batsmanId: 101,
    nonStrikerId: 102,
    bowlerId: 201,
    fielderId: 205,          // Fielder who ran out
    outPlayerId: 102,        // Non-striker ran out
    
    event: "runout",
    eventType: "wicket",
    dismissalType: "runout",
    extraType: "no-ball",    // ✅ It was a no-ball
    isLegal: false,          // ❌ No-ball is illegal
    runsOnThisBall: 1,       // They attempted 1 run
    extrasThisBall: 1,       // 1 no-ball penalty
    four: false,
    six: false
}
